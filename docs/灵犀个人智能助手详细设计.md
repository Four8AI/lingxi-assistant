    def get_skill(self, skill_id: str) -> dict:
        """获取技能信息"""
        return self.skill_cache.get(skill_id)
    
    def call(self, skill_id: str, parameters: dict) -> dict:
        """调用技能"""
        skill = self.get_skill(skill_id)
        if not skill:
            return {"success": False, "error": f"技能{skill_id}不存在"}
        
        if not skill["enabled"]:
            return {"success": False, "error": f"技能{skill_id}已禁用"}
        
        # 简化调用逻辑
        try:
            headers = {}
            if skill["api_key"]:
                headers["Authorization"] = f"Bearer {skill['api_key']}"
            
            response = requests.post(
                skill["api_url"],
                json=parameters,
                headers=headers,
                timeout=30
            )
            
            if response.status_code == 200:
                return {"success": True, "data": response.json()}
            else:
                return {"success": False, "error": f"HTTP {response.status_code}: {response.text}"}
                
        except Exception as e:
            return {"success": False, "error": str(e)}
    
    def list_skills(self) -> list:
        """列出所有可用技能"""
        return [
            {"skill_id": s["skill_id"], "skill_name": s["skill_name"], 
             "description": s["description"]}
            for s in self.skill_cache.values()
        ]

def process_task(task: str, session_id: str, config: dict, 
                 classifier, skill_registry, session_manager) -> str:
    """处理单个任务"""
    # 1. 任务分类
    classification = classifier.classify(task)
    level = classification["level"]
    log_info(f"任务分类：{level} (置信度：{classification['confidence']})")
    
    # 2. 获取会话历史
    history = session_manager.get_history(session_id)
    
    # 3. 选择执行引擎
    if level == "trivial":
        # 直接回答
        llm = LLMClient(config.llm)
        result = llm.generate(task, max_tokens=1000)
    elif level == "simple":
        # ReAct模式
        engine = ReActEngine(config, skill_registry)
        result_dict = engine.execute(task, session_id, history)
        result = result_dict.get("result", result_dict.get("error", "执行失败"))
    else:  # complex
        # Plan+ReAct模式
        engine = PlanReActEngine(config, skill_registry, session_manager)
        result_dict = engine.execute(task, session_id, history)
        result = result_dict.get("result", result_dict.get("error", "执行失败"))
    
    # 4. 保存会话
    session_manager.add_turn(session_id, "user", task)
    session_manager.add_turn(session_id, "assistant", result)
    
    # 5. 记录执行日志
    log_execution(task, level, result, session_id)
    
    return result

七、断点重试功能说明

7.1 功能概述

Plan+ReAct模式支持完整的步骤断点重试功能，确保在系统崩溃、网络中断或其他异常情况下，可以从上次失败的位置继续执行，而不是重新开始整个任务。

7.2 检查点机制

**检查点数据结构：**
```python
checkpoint = {
    "task": "原始任务描述",
    "plan": [步骤1, 步骤2, ...],           # 执行计划
    "current_step_idx": 2,                 # 当前步骤索引
    "completed_steps": [步骤1, 步骤2],      # 已完成的步骤
    "step_results": [结果1, 结果2],        # 步骤执行结果
    "replan_count": 1,                     # 重规划次数
    "execution_status": "running",         # 执行状态
    "error_info": None,                    # 错误信息
    "timestamp": 1234567890.0             # 时间戳
}
```

**执行状态：**
- `running`: 正在执行
- `replanned`: 已重规划
- `completed`: 执行完成
- `failed`: 执行失败

7.3 自动断点恢复

系统在执行Plan+ReAct任务时会自动：
1. 在开始执行时保存初始检查点
2. 每完成一个步骤后更新检查点
3. 在重规划时保存新的Plan和状态
4. 在任务完成时标记为completed状态

当任务中断后重新执行相同任务时，系统会：
1. 检查是否存在该任务的检查点
2. 如果存在且任务匹配，从上次失败的位置继续执行
3. 如果不存在或任务不匹配，开始新任务执行

**场景1：网络中断恢复**

```

**场景2：步骤失败重试**
```
用户：查询天气并翻译成英文
系统：步骤1/2：查询天气...成功
系统：步骤2/2：翻译...失败
系统：重试步骤2（1/3）...失败
系统：重试步骤2（2/3）...失败
系统：重试步骤2（3/3）...成功

```

**场景3：重规划恢复**
```
用户：规划北京到上海的旅行
系统：生成Plan（5个步骤）
系统：步骤1/5：查询航班...失败

系统：生成新Plan（4个步骤，改用高铁）
系统：继续执行新Plan...
```

7.7 注意事项

1. **检查点过期**：默认24小时后自动清理，可通过配置调整
2. **任务匹配**：只有任务描述完全相同时才会恢复检查点
3. **状态一致性**：重规划会生成新的Plan，已完成的步骤结果会保留
4. **资源清理**：定期清理过期检查点，避免占用过多存储空间
5. **并发控制**：同一会话同一时间只能有一个活跃的检查点

## 8.1 技能系统概述

灵犀支持通过扫描目录自动注册技能，无需手动代码注册或数据库操作。系统支持两种技能格式和三种技能类型：

### 技能格式
- **传统格式**：目录以 `Skill` 结尾，配置文件为 `skill.json`
- **MCP格式**：目录包含 `SKILL.md` 文件（YAML frontmatter格式）

### 技能类型
- **api**：调用外部API的技能
- **local**：本地执行的技能
- **mcp**：遵循MCP协议的技能

### 扫描目录
系统启动时自动扫描以下目录：
- 内置技能目录：`lingxi/skills/builtin`
- 用户技能目录：`.lingxi/skills`

### 目录结构

## 8.4 本地技能实现

```python
# FileReadSkill/main.py
def execute(parameters: dict) -> dict:
    """执行文件读取操作"""
    file_path = parameters.get("file_path")
    try:
        with open(file_path, "r", encoding="utf-8") as f:
            content = f.read()
        return {"success": True, "content": content}
    except Exception as e:
        return {"success": False, "error": str(e)}
```

# AdvancedSkill/main.py
import requests

def init():
    """初始化技能（系统启动时调用，可选）"""

def execute(parameters: dict) -> dict:

九、事件驱动架构（V4.0新增）

9.1 事件系统概述

V4.0版本引入事件驱动架构，统一处理执行过程中的各类事件，解耦输出逻辑：

```
执行引擎 → global_event_publisher → 订阅者
                                        ├── ConsoleSubscriber（控制台输出）
                                        ├── WebSocketSubscriber（前端推送）
                                        └── SessionStoreSubscriber（会话存储）
```

9.2 事件类型

| 事件名称 | 触发时机 | 携带数据 |
|---------|---------|---------|
| think_start | 思考开始 | session_id, execution_id |
| think_stream | 思考流式输出 | session_id, execution_id, content |
| think_final | 思考完成 | session_id, execution_id, thought |
| plan_start | 计划开始 | session_id, execution_id |
| plan_final | 计划完成 | session_id, execution_id, plan |
| step_start | 步骤开始 | session_id, execution_id, step_index, description |
| step_end | 步骤结束 | session_id, execution_id, step_index, result, status |
| task_start | 任务开始 | session_id, execution_id, task_input |
| task_end | 任务结束 | session_id, execution_id, result |
| task_failed | 任务失败 | session_id, execution_id, error |
