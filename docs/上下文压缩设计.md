├─────────────────────────────────────────────────────────────────┤

2.2 内容保留优先级
内容类型   保留策略   理由
用户输入   ✅ 完整保留（最近10轮）   用户意图最准确表达

关键实体   ✅ 完整保留   人名、地名、时间、数字等

任务目标   ✅ 完整保留   防止任务漂移

工具调用结果   ⚠️ 摘要保留   只保留关键数据，移除冗余

模型推理过程   ❌ 压缩/移除   Think过程可丢弃

已完成任务历史   ⚠️ 摘要归档   压缩后存入长期记忆

错误/重试记录   ❌ 移除   除非用于调试

class ContentType(Enum):

    ASSISTANT_RESPONSE = "assistant_response"

    TOOL_RESULT = "tool_result"

@dataclass
class ContextMessage:

class ContextManager:
    """上下文管理器 - 支持动态压缩"""
    
    def init(self, config: dict, session_id: str):
        self.config = config
        self.session_id = session_id
        self.messages: List[ContextMessage] = []
        self.current_task_id: Optional[str] = None
        self.token_usage = 0
        self.max_tokens = config.context_management.token_budget.max_tokens
        self.compression_trigger = config.context_management.token_budget.compression_trigger
        
        # 初始化长期记忆
        if config.context_management.long_term_memory.enabled:
            self.long_term_memory = LongTermMemory(
                db_path="data/long_term_memory.db",
                vector_dim=config.context_management.long_term_memory.vector_dim
            )
        
        # 加载持久化上下文
        self._load_from_db()
    
    def add_message(self, role: str, content: str, 
                    content_type: ContentType = None,
                    task_id: str = None,
                    metadata: Dict = None) -> ContextMessage:
        """添加消息到上下文"""
        # 估算token数（简化版，实际可用tiktoken）
        token_count = len(content) // 4
        
        message = ContextMessage(
            id=f"msg_{int(time.time() * 1000)}",
            role=role,
            content=content,
            content_type=content_type or self._infer_content_type(role, content),
            token_count=token_count,
            task_id=task_id or self.current_task_id,
            metadata=metadata or {}
        )
        
        self.messages.append(message)
        self.token_usage += token_count
        
        # 检查是否需要压缩
        if self._should_compress():
            self.compress()
        
        return message
    
    def _should_compress(self) -> bool:
        """判断是否需要压缩"""
        usage_ratio = self.token_usage / self.max_tokens
        return usage_ratio >= self.compression_trigger
    
    def compress(self, strategy: str = None) -> Dict:
        """执行上下文压缩"""
        strategy = strategy or self.config.context_management.compression.strategy
        
        stats = {
            "before_tokens": self.token_usage,
            "compressed_count": 0,
            "archived_count": 0
        }
        
        if strategy == "hybrid":
            # 混合策略：多种压缩方式组合
            stats.update(self._compress_thinking())      # 移除推理过程
            stats.update(self._compress_tool_results())  # 摘要工具结果
            stats.update(self._archive_old_tasks())      # 归档旧任务
            stats.update(self._sliding_window())         # 滑动窗口
            
        elif strategy == "summary":
            # LLM智能摘要
            stats.update(self._llm_summary())
            
        elif strategy == "sliding_window":
            # 简单滑动窗口
            stats.update(self._sliding_window())
        
        stats["after_tokens"] = self.token_usage
        stats["compression_ratio"] = 1 - (stats["after_tokens"] / stats["before_tokens"])
        
        return stats
    
    def _compress_thinking(self) -> Dict:
        """压缩策略1：移除模型推理过程（Think）"""
        compressed_count = 0
        removed_tokens = 0
        
        # 标记所有thinking类型为可删除
        for msg in self.messages:
            if msg.content_type == ContentType.THINKING:
                msg.is_compressed = True
                msg.summary = "[推理过程已压缩]"
                removed_tokens += msg.token_count
                compressed_count += 1
        
        # 从活跃上下文中移除（但保留在数据库用于审计）
        self.messages = [m for m in self.messages if m.content_type != ContentType.THINKING]
        self.token_usage -= removed_tokens
        
        return {"thinking_compressed": compressed_count, "tokens_freed": removed_tokens}
    
    def _compress_tool_results(self) -> Dict:
        """压缩策略2：摘要工具调用结果"""
        compressed_count = 0
        tokens_freed = 0
        
        # 找出可压缩的工具结果（保留最近N个）
        keep_count = self.config.context_management.retention.tool_result_keep_turns
        tool_results = [m for m in self.messages if m.content_type == ContentType.TOOL_RESULT]
        
        if len(tool_results) > keep_count:
            # 对旧的工具结果进行摘要
            to_compress = tool_results[:-keep_count]
            
            for msg in to_compress:
                if not msg.is_compressed:
                    # 使用LLM摘要或直接截断
                    if self.config.context_management.compression.enable_llm_summary:
                        msg.summary = self._summarize_with_llm(msg.content)
                    else:
                        msg.summary = self._truncate_summary(msg.content)
                    
                    msg.is_compressed = True
                    tokens_freed += int(msg.token_count * 0.7)  # 估算节省
                    compressed_count += 1
        
        return {"tool_results_compressed": compressed_count, "tokens_freed": tokens_freed}
    
    def _archive_old_tasks(self) -> Dict:
        """压缩策略3：归档已完成任务的历史"""
        archived_count = 0
        tokens_freed = 0
        
        if not self.config.context_management.retention.task_boundary_archive:
            return {"tasks_archived": 0, "tokens_freed": 0}
        
        # 找出已完成任务的消息
        current_task_msgs = [m for m in self.messages if m.task_id == self.current_task_id]
        old_task_msgs = [m for m in self.messages if m.task_id and m.task_id != self.current_task_id]
        
        if old_task_msgs:
            # 生成任务摘要
            task_summary = self._generate_task_summary(old_task_msgs)
            
            # 归档到长期记忆
            self.long_term_memory.store(
                task_id=old_task_msgs[0].task_id,
                summary=task_summary,
                messages=old_task_msgs
            )
            
            # 从活跃上下文移除
            archived_count = len(old_task_msgs)
            tokens_freed = sum(m.token_count for m in old_task_msgs)
            self.messages = current_task_msgs + [m for m in self.messages if not m.task_id]
            self.token_usage -= tokens_freed
        
        return {"tasks_archived": archived_count, "tokens_freed": tokens_freed}
    
    def _sliding_window(self) -> Dict:
        """压缩策略4：滑动窗口保留"""
        tokens_freed = 0
        
        # 用户输入保留最近N轮
        keep_turns = self.config.context_management.retention.user_input_keep_turns
        user_inputs = [m for m in self.messages if m.content_type == ContentType.USER_INPUT]
        
        if len(user_inputs) > keep_turns:

                if not msg.is_compressed:
                    msg.is_compressed = True
                    msg.summary = "[历史对话已归档]"
                    tokens_freed += msg.token_count
            
            self.token_usage -= tokens_freed
        
        return {"sliding_window_applied": True, "tokens_freed": tokens_freed}

import sqlite3
import json

from typing import List, Dict, Optional
from datetime import datetime

class LongTermMemory:
    """长期记忆存储 - SQLite + 简单向量检索"""
    
    def init(self, db_path: str = "data/long_term_memory.db", 
                 vector_dim: int = 384):
        self.db_path = db_path
        self.vector_dim = vector_dim
        self._init_db()
    
    def _init_db(self):
        """初始化数据库"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 任务摘要表
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS task_summaries (
                task_id TEXT PRIMARY KEY,
                session_id TEXT,
                summary TEXT,
                key_entities TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                accessed_at TIMESTAMP,
                access_count INTEGER DEFAULT 0
            )
        """)
        
        # 消息归档表
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS archived_messages (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                task_id TEXT,
                message_id TEXT,
                role TEXT,
                content_summary TEXT,
                content_type TEXT,
                token_count INTEGER,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        
        # 简单向量存储表（生产环境建议用Chroma/Milvus）
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS message_embeddings (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                task_id TEXT,
                embedding BLOB,
                content_preview TEXT
            )
        """)
        
        conn.commit()
        conn.close()
    
    def store(self, task_id: str, summary: str, messages: List, 
              session_id: str = "default"):
        """存储任务到长期记忆"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 存储任务摘要
        key_entities = self._extract_entities(summary)
        cursor.execute("""
            INSERT OR REPLACE INTO task_summaries 
            (task_id, session_id, summary, key_entities)
            VALUES (?, ?, ?, ?)
        """, (task_id, session_id, summary, json.dumps(key_entities)))
        
        # 归档消息
        for msg in messages:
            cursor.execute("""
                INSERT INTO archived_messages 
                (task_id, message_id, role, content_summary, content_type, token_count)
                VALUES (?, ?, ?, ?, ?, ?)
            """, (
                task_id, 
                msg.id, 
                msg.role, 
                msg.summary or msg.content[:200],
                msg.content_type.value,
                msg.token_count
            ))
        
        conn.commit()
        conn.close()
    
    def retrieve(self, query: str, top_k: int = 5) -> List[Dict]:
        """检索相关历史记忆"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 简单关键词检索（生产环境应用向量检索）
        keywords = query.split()
        results = []
        
        for keyword in keywords:
            cursor.execute("""
                SELECT task_id, summary, key_entities, access_count
                FROM task_summaries
                WHERE summary LIKE ? OR key_entities LIKE ?
                ORDER BY access_count DESC
                LIMIT ?
            """, (f"%{keyword}%", f"%{keyword}%", top_k))
            
            for row in cursor.fetchall():
                if row[0] not in [r["task_id"] for r in results]:
                    results.append({
                        "task_id": row[0],
                        "summary": row[1],
                        "key_entities": json.loads(row[2]) if row[2] else [],
                        "access_count": row[3]
                    })
        
        # 更新访问计数
        for r in results[:top_k]:
            cursor.execute("""
                UPDATE task_summaries 
                SET accessed_at = CURRENT_TIMESTAMP, access_count = access_count + 1
                WHERE task_id = ?
            """, (r["task_id"],))
        
        conn.commit()
        conn.close()
        
        return results[:top_k]

## V3.0 更新（2026-03-12）

### 问题诊断

#### 原压缩功能不生效的原因分析

在 V2.2 版本实现后，发现上下文压缩功能在实际运行中并未按预期工作。经过深入排查，定位到以下核心问题：

1. **ContextManager 与执行流程脱节**
   - `ContextManager` 类已实现，但未在 `TaskExecutor` 中实际使用
   - 执行流程仍直接构建上下文列表，绕过了压缩管理器
   - 缺少统一的上下文获取入口

2. **压缩触发条件过于宽松**
   - 原 `compression_trigger: 0.7` 在实际对话中很少达到
   - 缺少强制压缩机制，导致 token 使用率持续累积

3. **压缩策略单一**
   - 仅支持简单的滑动窗口，缺少智能摘要
   - 未区分内容类型进行差异化压缩

#### 根本原因


### 新方案：分层压缩策略

针对 V2.2 的问题，V3.0 引入**分层压缩策略**，在保证关键信息不丢失的前提下，最大化压缩效率。

#### 压缩层级

| 层级 | 策略 | 适用场景 | 压缩率 |
|------|------|---------|--------|
| **L1 轻量级压缩**（默认） | 移除推理过程 + 截断工具结果 | 日常对话 | 30-40% |
| **L2 LLM 智能压缩**（可选） | 使用 LLM 摘要历史对话 | 长对话/复杂任务 | 50-70% |
| **L3 归档压缩** | 将旧任务移至长期记忆 | 多任务场景 | 60-80% |

#### _compress_history() 双层压缩逻辑

```python
# src/session/manager.py

class SessionManager:
    # ... 现有代码 ...
    
    async def _compress_history(
        self,
        history: List[Dict[str, Any]],
        strategy: str = "lightweight"
    ) -> List[Dict[str, Any]]:
        """执行历史压缩（双层逻辑）
        
        第一层：轻量级压缩（快速，规则基础）
        第二层：LLM 智能压缩（慢速，语义理解）
        
        Args:
            history: 原始历史记录
            strategy: 压缩策略
        
        Returns:
            压缩后的历史
        """
        # ========== 第一层：轻量级压缩 ==========
        if strategy in ("lightweight", "llm_summary", "archived"):
            history = self._lightweight_compress(history)
        
        # 检查是否已达到压缩目标
        if self._token_count(history) < self.max_tokens * 0.5:
            return history
        
        # ========== 第二层：LLM 智能压缩 ==========
        if strategy == "llm_summary":
            try:
                history = await self._llm_summary_compress(history)
            except Exception as e:
                self.logger.warning(f"LLM 压缩失败，降级到滑动窗口：{e}")
                # 降级处理
                history = self._sliding_window_compress(history)
        
        elif strategy == "archived":
            history = self._archive_old_tasks(history)
        
        return history
    
    def _lightweight_compress(self, history: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """轻量级压缩：移除推理过程 + 截断工具结果
        
        快速规则基础压缩，不依赖 LLM
        """
        compressed = []
        
        for msg in history:
            # 跳过推理过程
            if msg.get("role") == "assistant" and msg.get("content", "").startswith("Think:"):
                continue
            
            # 截断过长的工具结果
            if msg.get("role") == "tool" and len(msg.get("content", "")) > 200:
                msg = msg.copy()
                msg["content"] = msg["content"][:200] + "...[截断]"
            
            compressed.append(msg)
        
        return compressed
    
    async def _llm_summary_compress(self, history: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """LLM 智能压缩：使用 LLM 摘要历史对话
        
        保留关键信息，压缩冗余内容
        """
        # 分组：每 5 轮对话为一组进行摘要
        groups = [history[i:i+10] for i in range(0, len(history), 10)]
        compressed = []
        
        for i, group in enumerate(groups):
            if i == len(groups) - 1:
                # 最后一组保留完整（最近对话）
                compressed.extend(group)
            else:
                # 其他组进行 LLM 摘要
                summary = await self._summarize_group(group)
                compressed.append({
                    "role": "system",
                    "content": f"[历史对话摘要] {summary}"
                })
        
        return compressed
    
    async def _summarize_group(self, messages: List[Dict[str, Any]]) -> str:
        """使用 LLM 摘要一组对话"""
        # 构建摘要提示
        conversation = "\n".join([
            f"{m['role']}: {m['content'][:100]}"
            for m in messages
        ])
        
        prompt = f"""请摘要以下对话，保留关键信息（人名、时间、任务目标、结果）：

{conversation}

摘要（50 字以内）："""
        
        # 调用 LLM（使用低优先级模型以节省成本）
        response = await self.llm_client.generate(
            messages=[{"role": "user", "content": prompt}],
            max_tokens=100,
            temperature=0.3
        )
        
        return response.strip()
    
    def _sliding_window_compress(self, history: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """滑动窗口压缩：保留最近 N 轮
        
        降级方案，当 LLM 压缩失败时使用
        """
        keep_turns = self.config.context_management.retention.user_input_keep_turns
        user_inputs = [m for m in history if m.get("role") == "user"]
        
        if len(user_inputs) <= keep_turns:
            return history
        
        # 保留最近 N 轮用户输入对应的对话
        recent_inputs = user_inputs[-keep_turns:]
        compressed = []
        
        for msg in history:
            if msg.get("role") == "user" and msg not in recent_inputs:
                continue
            compressed.append(msg)
        
        return compressed
    
    def _archive_old_tasks(self, history: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """归档旧任务到长期记忆"""
        # 识别任务边界（基于系统消息或任务 ID）
        current_task_msgs = []
        old_task_msgs = []
        
        for msg in history:
            if msg.get("metadata", {}).get("task_id"):
                old_task_msgs.append(msg)
            else:
                current_task_msgs.append(msg)
        
        # 归档旧任务
        if old_task_msgs:
            await self._archive_to_long_term_memory(old_task_msgs)
        
        return current_task_msgs
    
    async def _archive_to_long_term_memory(self, messages: List[Dict[str, Any]]):
        """将消息归档到长期记忆数据库"""
        # 实现归档逻辑
        pass
    
    def _token_count(self, messages: List[Dict[str, Any]]) -> int:
        """估算 token 数量"""
        total = 0
        for msg in messages:
            content = msg.get("content", "")
            total += len(content) // 4  # 简单估算：4 字符 ≈ 1 token
        return total
```

#### 降级处理机制

```python
# src/session/manager.py

class SessionManager:
    # ... 现有代码 ...
    
    async def get_history_with_fallback(
        self,
        session_id: str,
        target_token_limit: int = None
    ) -> List[Dict[str, Any]]:
        """获取历史（带降级处理）
        
        压缩策略降级链：
        LLM 智能压缩 → 轻量级压缩 → 滑动窗口 → 强制截断
        
        Args:
            session_id: 会话 ID
            target_token_limit: 目标 token 限制
        
        Returns:
            压缩后的历史
        """
        target_token_limit = target_token_limit or self.max_tokens * 0.5
        
        # 尝试策略 1: LLM 智能压缩
        try:
            history = await self.get_history(
                session_id,
                compress=True,
                compression_strategy="llm_summary"
            )
            if self._token_count(history) <= target_token_limit:
                return history
        except Exception as e:
            self.logger.warning(f"LLM 压缩失败：{e}")
        
        # 降级到策略 2: 轻量级压缩
        try:
            history = await self.get_history(
                session_id,
                compress=True,
                compression_strategy="lightweight"
            )
            if self._token_count(history) <= target_token_limit:
                return history
        except Exception as e:
            self.logger.warning(f"轻量级压缩失败：{e}")
        
        # 降级到策略 3: 滑动窗口
        try:
            history = await self.get_history(
                session_id,
                compress=True,
                compression_strategy="archived"
            )
            if self._token_count(history) <= target_token_limit:
                return history
        except Exception as e:
            self.logger.warning(f"归档压缩失败：{e}")
        
        # 最后手段：强制截断
        self.logger.error("所有压缩策略失败，使用强制截断")
        history = await self.get_history(session_id, limit=5, compress=False)
        return history
```
